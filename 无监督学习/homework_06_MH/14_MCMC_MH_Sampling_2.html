def sampling(theta_curr, x, n):
    theta_cand = beta.rvs(a=1,b=1)
    accept_ratio = min(0, x*np.log(theta_cand)+(n-x)*np.log(1-theta_cand)-x*np.log(theta_curr)-(n-x)*np.log(1-theta_curr))
    u = np.random.rand()
    if u==0 or np.log(u) <= accept_ratio:
        return theta_cand
    else:
        return theta_curr

n = 10
x = 6
n_iter = 20000
theta = -10.0
theta_draws = []
for i in range(n_iter):
    theta = sampling(theta, x, n)
    theta_draws.append(theta)
theta_draws = np.array(theta_draws)

plt.plot(range(n_iter), theta_draws)
plt.xlabel('$\\theta$')
plt.show()

plt.hist(theta_draws[int(n_iter/2):], edgecolor = 'r', bins=50)
plt.show()

plot_acf(theta_draws[int(n_iter/2):],lags=50)
plt.show()

print("True Mean: %f" % beta.mean(a=7, b=5))
print("Estimated Mean: %f" % np.mean(theta_draws[int(n_iter/2):]))
print("True Var: %f" % beta.var(a=7, b=5))
print("Estimated Var: %f" % np.var(theta_draws[int(n_iter/2):]))
